CREATE OR REPLACE FUNCTION trg_after_student_degree_insert()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO financial_transations (
    transation_value,
    transation_date,
    type,
    student_person_id
  )
  VALUES (
    (SELECT degree_value
     FROM degree
     WHERE id = NEW.degree_id),
    NEW.enrollement_date,
    'degree_enrollment',
    NEW.student_person_id
  );

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;


DROP TRIGGER IF EXISTS trg_student_degree_insert
  ON student_degree;


CREATE TRIGGER trg_student_degree_insert
AFTER INSERT ON student_degree
FOR EACH ROW
EXECUTE FUNCTION trg_after_student_degree_insert();



CREATE OR REPLACE FUNCTION trg_after_student_activity_insert()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO financial_transations (
    transation_value,
    transation_date,
    type,
    student_person_id
  )
  VALUES (
    (SELECT activity_value
     FROM extracurricular_activities
     WHERE id = NEW.extracurricular_activities_id),
    CURRENT_DATE,            
    'activitie_enrollment',   
    NEW.student_person_id    
  );

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;


DROP TRIGGER IF EXISTS trg_after_student_activity_insert
  ON student_extracurricular_activities;

CREATE TRIGGER trg_after_student_activity_insert
AFTER INSERT ON student_extracurricular_activities
FOR EACH ROW
EXECUTE FUNCTION trg_after_student_activity_insert();


CREATE OR REPLACE FUNCTION decrement_course_edition_capacity()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
  -- Atualiza a edição de curso correspondente à turma inserida
  UPDATE course_edition ce
  SET capacity = ce.capacity - 1
  FROM timetable t
  WHERE t.class_class_number = NEW.class_class_number
    AND ce.edition_id   = t.course_edition_edition_id
    AND ce.course_course_code = t.course_edition_course_course_code;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_decrement_capacity
AFTER INSERT ON student_class
FOR EACH ROW
EXECUTE FUNCTION decrement_course_edition_capacity();
